        function setLanguage(lang) {
            document.querySelectorAll('[data-key]').forEach(element => {
                const key = element.getAttribute('data-key');
                if (translations[lang] && translations[lang][key]) {
                    // Handle special cases for HTML content within text
                    if (key === "jobs-intro" || key === "schemes-intro" || key === "footer-copyright" || key === "footer-support" || key === "admission-intro" || key === "scholarship-intro") {
                        const tempDiv = document.createElement('div');
                        tempDiv.innerHTML = translations[lang][key];
                        // Preserve original links, only update text nodes
                        Array.from(element.childNodes).forEach(node => {
                            if (node.nodeType === Node.TEXT_NODE) {
                                node.textContent = tempDiv.textContent; // Simple text content update
                            } else if (node.nodeType === Node.ELEMENT_NODE && node.tagName === 'A') {
                                // For anchor tags, ensure the text content is updated if it's part of the translation
                                const originalHref = node.getAttribute('href');
                                const originalTarget = node.getAttribute('target');
                                const originalText = node.textContent;
                                
                                // Find the corresponding text in the translated string
                                const regex = new RegExp(`>([^<]+)<`, 'g');
                                let match;
                                let newText = originalText;
                                while ((match = regex.exec(tempDiv.innerHTML)) !== null) {
                                    if (originalText.includes(match[1])) { // Simple check, might need more robust logic for complex cases
                                        newText = match[1];
                                        break;
                                    }
                                }
                                node.textContent = newText; // Update text content of the link
                                node.setAttribute('href', originalHref); // Preserve original href
                                node.setAttribute('target', originalTarget); // Preserve original target
                            } else {
                                element.innerHTML = translations[lang][key]; // Fallback for complex HTML structures
                            }
                        });
                         if (element.children.length === 0 && element.textContent.trim() === '') {
                            element.innerHTML = translations[lang][key]; // If no children and empty, set innerHTML
                        }

                    } else if (key === "search-placeholder" || key === "summarizer-placeholder") {
                        element.setAttribute('placeholder', translations[lang][key]);
                    } else {
                        element.textContent = translations[lang][key];
                    }
                }
            });
            // Update document title separately
            document.title = translations[lang]["main-title"] + " - Your Ultimate Government Update Portal";
            // Update the text next to the logo as it's now a span
            const logoTextSpan = document.querySelector('.logo-container span');
            if (logoTextSpan) {
                logoTextSpan.textContent = translations[lang]["main-title"];
            }
        }

                    });

                    const result = await response.json();

                    if (result.candidates && result.candidates.length > 0 &&
                        result.candidates[0].content && result.candidates[0].content.parts &&
                        result.candidates[0].content.parts.length > 0) {
                        const summary = result.candidates[0].content.parts[0].text;
                        summaryText.textContent = summary;
                        summaryOutput.classList.remove('hidden');
                    } else {
                        summarizerError.textContent = translations[languageSelector.value]["error-message"] + " (No summary generated.)";
                        summarizerError.classList.remove('hidden');
                    }
                } catch (error) {
                    console.error('Error calling Gemini API:', error);
                    summarizerError.textContent = translations[languageSelector.value]["error-message"] + ` (${error.message || 'Unknown error'})`;
                    summarizerError.classList.remove('hidden');
                } finally {
                    summarizerLoading.classList.add('hidden');
                }
            });
        });
    </script>
</body>
